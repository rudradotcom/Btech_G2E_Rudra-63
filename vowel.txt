#include <stdio.h>
#include <ctype.h>

int main() {
    char ch;

    printf("Enter a character: ");
    scanf(" %c", &ch);

    // Convert to lowercase for easy comparison
    ch = tolower(ch);

    if (isalpha(ch)) {
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
            printf("It is a Vowel.\n");
        } else {
            printf("It is a Consonant.\n");
        }
    } else {
        printf("It is not an alphabet character.\n");
    }

    return 0;
}
